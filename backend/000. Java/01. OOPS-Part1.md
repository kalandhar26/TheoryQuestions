### Is-a Relationship (Inheritance)

- The "is-a" relationship represents inheritance, where a class inherits from another class. It means the child class is
  a type of the parent class.

```java
// Parent class
class Animal {
    protected String name;
    protected int age;

    public Animal(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public void eat() {
        System.out.println(name + " is eating");
    }

    public void sleep() {
        System.out.println(name + " is sleeping");
    }
}

// Child class - Dog "is-a" Animal
class Dog extends Animal {
    private String breed;

    public Dog(String name, int age, String breed) {
        super(name, age);
        this.breed = breed;
    }

    public void bark() {
        System.out.println(name + " is barking");
    }

    @Override
    public void eat() {
        System.out.println(name + " is eating dog food");
    }
}

// Another child class - Cat "is-a" Animal
class Cat extends Animal {
    private boolean isIndoor;

    public Cat(String name, int age, boolean isIndoor) {
        super(name, age);
        this.isIndoor = isIndoor;
    }

    public void meow() {
        System.out.println(name + " is meowing");
    }
}

// Usage example
public class IsADemo {
    public static void main(String[] args) {
        Dog dog = new Dog("Buddy", 3, "Golden Retriever");
        Cat cat = new Cat("Whiskers", 2, true);

        // Dog is-a Animal
        dog.eat();      // Inherited method
        dog.sleep();    // Inherited method
        dog.bark();     // Dog-specific method

        // Cat is-a Animal  
        cat.eat();      // Inherited method
        cat.sleep();    // Inherited method
        cat.meow();     // Cat-specific method
    }
}
```

### Has-a Relationship (Composition)

- The "has-a" relationship represents composition, where a class contains objects of other classes as members.

```java
// Component classes
class Engine {
    private String type;
    private int horsepower;

    public Engine(String type, int horsepower) {
        this.type = type;
        this.horsepower = horsepower;
    }

    public void start() {
        System.out.println(type + " engine with " + horsepower + "hp started");
    }

    public void stop() {
        System.out.println("Engine stopped");
    }
}

class Wheels {
    private int count;
    private String type;

    public Wheels(int count, String type) {
        this.count = count;
        this.type = type;
    }

    public void rotate() {
        System.out.println(count + " " + type + " wheels are rotating");
    }
}

// Car has-a Engine and has-a Wheels
class Car {
    private String model;
    private Engine engine;      // Has-a relationship
    private Wheels wheels;       // Has-a relationship

    public Car(String model, Engine engine, Wheels wheels) {
        this.model = model;
        this.engine = engine;    // Composition
        this.wheels = wheels;     // Composition
    }

    public void startCar() {
        System.out.println("Starting " + model);
        engine.start();           // Using composed object
        wheels.rotate();          // Using composed object
    }

    public void stopCar() {
        System.out.println("Stopping " + model);
        engine.stop();
    }
}

// Usage example
public class HasADemo {
    public static void main(String[] args) {
        Engine v8Engine = new Engine("V8", 450);
        Wheels alloyWheels = new Wheels(4, "Alloy");

        Car myCar = new Car("Mustang", v8Engine, alloyWheels);
        myCar.startCar();  // Car uses its components
        myCar.stopCar();
    }
}
```

### Association

- Association represents a relationship between two separate classes. It can be uni-directional or bi-directional.

```java
import java.util.ArrayList;
import java.util.List;

class Student {
    private String name;
    private List<Course> courses;  // Association with Course

    public Student(String name) {
        this.name = name;
        this.courses = new ArrayList<>();
    }

    public void enrollInCourse(Course course) {
        courses.add(course);
        course.addStudent(this);  // Bi-directional association
    }

    public String getName() {
        return name;
    }

    public void displayCourses() {
        System.out.println(name + " is enrolled in:");
        for (Course course : courses) {
            System.out.println("  - " + course.getName());
        }
    }
}

class Course {
    private String name;
    private List<Student> students;  // Association with Student

    public Course(String name) {
        this.name = name;
        this.students = new ArrayList<>();
    }

    public String getName() {
        return name;
    }

    public void addStudent(Student student) {
        students.add(student);
    }

    public void displayStudents() {
        System.out.println("Students in " + name + ":");
        for (Student student : students) {
            System.out.println("  - " + student.getName());
        }
    }
}

// Usage example
public class AssociationDemo {
    public static void main(String[] args) {
        // Create objects
        Student john = new Student("John");
        Student jane = new Student("Jane");

        Course math = new Course("Mathematics");
        Course physics = new Course("Physics");

        // Establish associations
        john.enrollInCourse(math);
        john.enrollInCourse(physics);
        jane.enrollInCourse(math);

        // Display relationships
        john.displayCourses();
        math.displayStudents();
    }
}
```

### Aggregation

- Aggregation is a special form of association where the contained objects can exist independently of the container.

```java
import java.util.ArrayList;
import java.util.List;

class Department {
    private String name;
    private List<Professor> professors;  // Aggregation

    public Department(String name) {
        this.name = name;
        this.professors = new ArrayList<>();
    }

    public void addProfessor(Professor professor) {
        professors.add(professor);
    }

    public void removeProfessor(Professor professor) {
        professors.remove(professor);
    }

    public void displayProfessors() {
        System.out.println("Professors in " + name + " department:");
        for (Professor prof : professors) {
            System.out.println("  - " + prof.getName());
        }
    }

    public String getName() {
        return name;
    }
}

class Professor {
    private String name;
    private String specialization;
    private List<Department> departments;  // Professor can be in multiple depts

    public Professor(String name, String specialization) {
        this.name = name;
        this.specialization = specialization;
        this.departments = new ArrayList<>();
    }

    public void assignToDepartment(Department dept) {
        departments.add(dept);
        dept.addProfessor(this);  // Bi-directional aggregation
    }

    public String getName() {
        return name;
    }

    public void displayDepartments() {
        System.out.println(name + " works in:");
        for (Department dept : departments) {
            System.out.println("  - " + dept.getName());
        }
    }
}

// Usage example
public class AggregationDemo {
    public static void main(String[] args) {
        // Create departments
        Department cs = new Department("Computer Science");
        Department math = new Department("Mathematics");

        // Create professors
        Professor alice = new Professor("Dr. Alice", "AI");
        Professor bob = new Professor("Dr. Bob", "Algorithms");

        // Establish aggregation relationships
        alice.assignToDepartment(cs);
        bob.assignToDepartment(cs);
        bob.assignToDepartment(math);  // Bob works in two departments

        // Display relationships
        cs.displayProfessors();
        bob.displayDepartments();

        // Note: Professors exist independently of departments
        // If CS department closes, professors still exist
        System.out.println("\nEven if CS department closes:");
        System.out.println(alice.getName() + " still exists as a professor");
    }
}
```

| Relationship | Type        | Description                     | Lifetime          | Example                   |
|--------------|-------------|---------------------------------|-------------------|---------------------------|
| Is-a         | Inheritance | Child class is a type of parent | Same as parent    | Dog is-a Animal           |
| Has-a        | Composition | Strong containment              | Same as container | Car has-a Engine          |
| Association  | Usage       | Objects know about each other   | Independent       | Student takes Course      |
| Aggregation  | Collection  | Weak containment                | Independent       | Department has Professors |

```text
Is-a (Inheritance):        Animal ←── Dog
                           (Parent)   (Child)

Has-a (Composition):       Car ──┬── Engine
                               ├── Wheels
                               └── Seats

Association:               Student ────→ Course
                           (uses/knows)

Aggregation:               Department ──◇── Professor
                           (contains)   (exists independently)
```

### Dependency (Uses-a Relationship)

- Dependency is the weakest form of relationship where one class uses another class temporarily, typically as method
  parameters or local variables.

```java
// Dependent class
class Printer {
    public void print(String content) {
        System.out.println("Printing: " + content);
    }
}

class Document {
    private String content;

    public Document(String content) {
        this.content = content;
    }

    // Dependency: Document uses Printer temporarily
    public void print(Printer printer) {  // Printer is passed as parameter
        printer.print(this.content);
    }

    public String getContent() {
        return content;
    }
}

// Another example with stronger dependency
class ReportGenerator {
    // Dependency through method parameter
    public void generateReport(DataFetcher fetcher, Formatter formatter) {
        Data data = fetcher.fetchData();      // Using fetcher
        String formatted = formatter.format(data);  // Using formatter
        System.out.println(formatted);
    }
}

class DataFetcher {
    public Data fetchData() {
        return new Data("Sample data");
    }
}

class Formatter {
    public String format(Data data) {
        return "Formatted: " + data.toString();
    }
}

class Data {
    private String content;

    public Data(String content) {
        this.content = content;
    }

    public String toString() {
        return content;
    }
}

// Usage
public class DependencyDemo {
    public static void main(String[] args) {
        Document doc = new Document("Hello World");
        Printer printer = new Printer();
        doc.print(printer);  // Document depends on Printer temporarily
    }
}
```

### Realization (Interface Implementation)

- Realization represents the relationship between an interface and the classes that implement it.

```java
// Interface (the contract)
interface Drawable {
    void draw();

    void resize(double factor);
}

interface Colorable {
    void setColor(String color);

    String getColor();
}

// Realization: Circle implements Drawable and Colorable
class Circle implements Drawable, Colorable {
    private double radius;
    private String color;

    public Circle(double radius) {
        this.radius = radius;
        this.color = "black";
    }

    @Override
    public void draw() {
        System.out.println("Drawing a " + color + " circle with radius " + radius);
    }

    @Override
    public void resize(double factor) {
        this.radius *= factor;
        System.out.println("Circle resized to radius " + radius);
    }

    @Override
    public void setColor(String color) {
        this.color = color;
    }

    @Override
    public String getColor() {
        return color;
    }
}

// Another realization
class Rectangle implements Drawable {
    private double width, height;
    private String color;

    public Rectangle(double width, double height) {
        this.width = width;
        this.height = height;
        this.color = "black";
    }

    @Override
    public void draw() {
        System.out.println("Drawing a " + color + " rectangle " + width + "x" + height);
    }

    @Override
    public void resize(double factor) {
        this.width *= factor;
        this.height *= factor;
        System.out.println("Rectangle resized to " + width + "x" + height);
    }

    public void setColor(String color) {
        this.color = color;
    }
}

// Usage
public class RealizationDemo {
    public static void main(String[] args) {
        Drawable circle = new Circle(5.0);
        Drawable rectangle = new Rectangle(4.0, 6.0);

        circle.draw();
        circle.resize(2.0);

        rectangle.draw();
        rectangle.resize(0.5);
    }
}
```

### Composition vs Aggregation - Detailed Comparison

```java
import java.util.ArrayList;
import java.util.List;

// STRONG COMPOSITION - Parts cannot exist without whole
class House {
    private String address;
    private List<Room> rooms;  // Composition: Rooms are part of House

    public House(String address) {
        this.address = address;
        this.rooms = new ArrayList<>();
        // Rooms are created WITH the house
        rooms.add(new Room("Living Room"));
        rooms.add(new Room("Bedroom"));
        rooms.add(new Room("Kitchen"));
    }

    public void describe() {
        System.out.println("House at " + address + " has:");
        for (Room room : rooms) {
            System.out.println("  - " + room.getName());
        }
    }

    // When House is destroyed, Rooms are destroyed too
    // No way to access Rooms independently

    // Inner class - strong composition
    class Room {
        private String name;

        public Room(String name) {
            this.name = name;
        }

        public String getName() {
            return name;
        }
    }
}

// WEAK AGGREGATION - Parts can exist independently
class University {
    private String name;
    private List<Student> students;  // Aggregation

    public University(String name) {
        this.name = name;
        this.students = new ArrayList<>();
    }

    public void addStudent(Student student) {
        students.add(student);
        student.setUniversity(this);  // Bidirectional
    }

    public void removeStudent(Student student) {
        students.remove(student);
        student.setUniversity(null);
    }

    public String getName() {
        return name;
    }

    public void listStudents() {
        System.out.println("Students at " + name + ":");
        for (Student student : students) {
            System.out.println("  - " + student.getName());
        }
    }
}

class Student {
    private String name;
    private University university;  // Aggregation reference

    public Student(String name) {
        this.name = name;
        // Student exists even without University
    }

    public void setUniversity(University university) {
        this.university = university;
    }

    public String getName() {
        return name;
    }

    public void study() {
        if (university != null) {
            System.out.println(name + " is studying at " + university.getName());
        } else {
            System.out.println(name + " is studying independently");
        }
    }
}

// Usage
public class CompositionVsAggregationDemo {
    public static void main(String[] args) {
        // Composition example
        House house = new House("123 Main St");
        house.describe();
        // When house is garbage collected, rooms are gone

        // Aggregation example
        University uni = new University("Tech University");
        Student alice = new Student("Alice");
        Student bob = new Student("Bob");

        // Students exist before university
        alice.study();  // Studying independently

        // Aggregation relationship
        uni.addStudent(alice);
        uni.addStudent(bob);

        uni.listStudents();
        alice.study();  // Now studying at university

        // University closes
        uni = null;  // University is gone
        // But students still exist!
        alice.study();  // Still can study independently
    }
}
```

### Multiplicity in Relationships

```java
import java.util.*;

// Demonstrating different multiplicities
class Team {
    private String name;
    private List<Player> players;      // One-to-Many
    private Coach headCoach;            // One-to-One
    private List<Tournament> tournaments; // Many-to-Many

    public Team(String name) {
        this.name = name;
        this.players = new ArrayList<>();
        this.tournaments = new ArrayList<>();
    }

    public void addPlayer(Player player) {
        players.add(player);
        player.setTeam(this);
    }

    public void setCoach(Coach coach) {
        this.headCoach = coach;
        coach.setTeam(this);
    }

    public void addTournament(Tournament tournament) {
        tournaments.add(tournament);
        tournament.addTeam(this);
    }
}

class Player {
    private String name;
    private Team team;  // Many-to-One

    public Player(String name) {
        this.name = name;
    }

    public void setTeam(Team team) {
        this.team = team;
    }
}

class Coach {
    private String name;
    private Team team;  // One-to-One

    public Coach(String name) {
        this.name = name;
    }

    public void setTeam(Team team) {
        this.team = team;
    }
}

class Tournament {
    private String name;
    private List<Team> teams;  // Many-to-Many

    public Tournament(String name) {
        this.name = name;
        this.teams = new ArrayList<>();
    }

    public void addTeam(Team team) {
        teams.add(team);
    }
}
```

### Summary Table of All Relationships

| Relationship        | Type           | Strength    | Lifetime    | UML Notation                  | Java Implementation               |
|---------------------|----------------|-------------|-------------|-------------------------------|-----------------------------------|
| Is-a                | Inheritance    | Strong      | Same        | Solid line with hollow arrow  | `extends`                         |
| Has-a (Composition) | Strongest      | Very Strong | Same        | Filled diamond                | Member variable + instantiation   |
| Aggregation         | Containment    | Medium      | Independent | Hollow diamond                | Member variable + setter          |
| Association         | Usage          | Weak        | Independent | Solid line arrow              | Method parameter / return         |
| Dependency          | Usage          | Weakest     | Temporary   | Dashed line arrow             | Local variable / method parameter |
| Realization         | Implementation | Strong      | Same        | Dashed line with hollow arrow | `implements`                      |

### Real-World Example Combining All Concepts

```java
// Complete example showing all relationships
interface Payable {
    double calculateSalary();
}

abstract class Person {
    protected String name;
    protected int age;

    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }

    public abstract void introduce();
}

class Employee extends Person implements Payable {  // Is-a + Realization
    private double baseSalary;
    private Department department;  // Association
    private List<Project> projects;  // Aggregation
    private ParkingSpot parkingSpot;  // Composition-like (optional)

    public Employee(String name, int age, double baseSalary) {
        super(name, age);
        this.baseSalary = baseSalary;
        this.projects = new ArrayList<>();
    }

    @Override
    public double calculateSalary() {  // Realization
        double bonus = projects.size() * 1000;
        return baseSalary + bonus;
    }

    @Override
    public void introduce() {  // Is-a implementation
        System.out.println("I'm employee " + name);
    }

    public void assignToDepartment(Department dept) {  // Association
        this.department = dept;
        dept.addEmployee(this);
    }

    public void addProject(Project project) {  // Aggregation
        projects.add(project);
        project.addTeamMember(this);
    }

    public void assignParkingSpot(ParkingSpot spot) {  // Dependency-like
        this.parkingSpot = spot;
        System.out.println(name + " assigned parking spot " + spot.getNumber());
    }

    public void work() {
        System.out.println(name + " is working on " + projects.size() + " projects");
    }
}

class Department {
    private String name;
    private List<Employee> employees;  // Aggregation

    public Department(String name) {
        this.name = name;
        this.employees = new ArrayList<>();
    }

    public void addEmployee(Employee emp) {
        employees.add(emp);
    }
}

class Project {
    private String name;
    private List<Employee> team;  // Aggregation

    public Project(String name) {
        this.name = name;
        this.team = new ArrayList<>();
    }

    public void addTeamMember(Employee emp) {
        team.add(emp);
    }
}

class ParkingSpot {
    private String number;

    public ParkingSpot(String number) {
        this.number = number;
    }

    public String getNumber() {
        return number;
    }
}

// Usage
public class CompleteExample {
    public static void main(String[] args) {
        // Create objects
        Department it = new Department("IT");
        Project projectX = new Project("Project X");
        ParkingSpot spotA1 = new ParkingSpot("A1");

        Employee emp = new Employee("John", 30, 50000);

        // Establish relationships
        emp.assignToDepartment(it);     // Association
        emp.addProject(projectX);        // Aggregation
        emp.assignParkingSpot(spotA1);   // Dependency-like

        // Use the objects
        emp.introduce();                  // Is-a
        System.out.println("Salary: $" + emp.calculateSalary());  // Realization
        emp.work();                       // Association usage
    }
}
```
